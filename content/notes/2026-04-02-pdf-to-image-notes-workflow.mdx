<code>package com.pdf2pic;

import org.apache.pdfbox.rendering.ImageType;

import java.io.File;

/**
 * PDF转图片工具
 * 用于将指定文件夹下的PDF文件转换为图片
 */
public class Pdf2PicConverter {

    public static void main(String[] args) {
        // 项目根目录下 java -jar target/pdf2pic-1.0-SNAPSHOT-jar-with-dependencies.jar pdfs output
        if (args.length < 2) {
            System.out.println(&#34;用法: java -jar pdf2pic.jar <PDF源文件夹> <图片输出文件夹> [DPI] [格式]&#34;);
            System.out.println(&#34;  [DPI]: 可选参数&#xff0c;图片DPI值&#xff0c;默认为300&#34;);
            System.out.println(&#34;  [格式]: 可选参数&#xff0c;输出图片格式&#xff0c;默认为png&#34;);
            return;
        }

        String pdfFolderPath &#61; args[0];
        String outputBasePath &#61; args[1];
        
        // 解析可选参数
        float dpi &#61; 300;
        String format &#61; &#34;png&#34;;
        
        if (args.length > 2) {
            try {
                dpi &#61; Float.parseFloat(args[2]);
            } catch (NumberFormatException e) {
                System.out.println(&#34;警告: 无效的DPI值&#xff0c;使用默认值300&#34;);
            }
        }
        
        if (args.length > 3) {
            format &#61; args[3];
        }

        try {
            System.out.println(&#34;PDF转图片工具启动...&#34;);
            System.out.println(&#34;PDF源文件夹: &#34; &#43; pdfFolderPath);
            System.out.println(&#34;图片输出文件夹: &#34; &#43; outputBasePath);
            System.out.println(&#34;图片DPI: &#34; &#43; dpi);
            System.out.println(&#34;图片格式: &#34; &#43; format);
            
            // 获取项目根目录下的水印图片
            File watermarkFile &#61; new File(&#34;2.png&#34;);
            if (watermarkFile.exists()) {
                System.out.println(&#34;水印图片: &#34; &#43; watermarkFile.getAbsolutePath());
            } else {
                System.out.println(&#34;警告: 未找到水印图片 2.png&#34;);
            }
            
            // 创建转换器并设置参数
            PdfConverter converter &#61; new PdfConverter()
                .withDpi(dpi)
                .withFormat(format)
                .withImageType(ImageType.RGB);
            
            // 设置水印图片&#xff08;如果存在&#xff09;
            if (watermarkFile.exists()) {
                converter.withWatermark(watermarkFile);
            }
            
            // 执行批量转换
            converter.batchConvert(pdfFolderPath, outputBasePath, true);
            
            System.out.println(&#34;所有PDF文件处理完成!&#34;);
            
        } catch (Exception e) {
            System.out.println(&#34;处理过程中发生错误:&#34;);
            e.printStackTrace();
        }
    }
}</code></pre>
